# 二、列表、元组

#slice语法
L=[1,2,3,4,5,6,'a','b','c']

#对单个元素赋多个值就是新增
L[1:2]=['q','w']
print(L)

#对多个元素赋单个值就是删除
# 切记这个时候不能放串,不然就拆开放了
L[1:3]="W"
print(L)

#输出列表所有方法
print(dir(L))

列表的其他方法：

LL=[1,2,2,3,2, 'W', 3, 4, 5, 6, 'a', 'b', 'c']
#统计元素出现个数
print(LL.count(2))

#元素索引，默认输出第一个
print(LL.index(2))
#输出所有：
for i in range(len(LL)):
    if LL[i]==2:
        print(i)

#栈操作默认弹出并删除！！！最后一个！！！元素，但是可以通过传参指定弹出的元素
x=LL.pop(0)
print(LL)

#逆序存放
LL.reverse()
print(LL)

#int排序,传参reverse=True➡倒排
INT_list=[1,3,4,6,4,2,6,3,9]
INT_list.sort(reverse=True)
print(INT_list)

#str排序，传参key=len,指定按长度排序
STR_list=['wang','bob','boooooom','runoob','deng']
STR_list.sort(key=len)
print(STR_list)
STR_list.sort()
print(STR_list)

# 三、字典

#列表元组创建字典

L_t=[('a',1),('b',2),('c',3)]
dict1=dict(L_t)
print(dict1)

T_l=([1,2],[3,4],[5,6])
dict2=dict(T_l)
print(dict2)

#⬆都行，但是蠢
#直接创建,此时key不需要加引号
dic=dict(spanish='xibanya',china='zhongguo',a=1)
print(dic)

#   {'spanish': 'xibanya', 'china': 'zhongguo', 'a': 1}

#访问value
print(dic['china'])
#添加,直接对未知key赋值
dic['japan']='GouRiDi'
print(dic)
#删除
del dic['japan']
print(dic)

#判断,只能判断索引
# print('a' in dic)#返回0，1
#value遍历，且for循环的i就是key
for i in dic:
   if dic[i]==1:
       print('true')


'''
列表不允许对不存在索引赋值，而字典可以（新建）
'''

方法：

# items()

#items() 方法把字典中每对 key 和 value 组成一个元组，并把这些元组放在列表中返回
# print(dic.items)
Item=dic.items()
#字符串的方式输出
print("%s" % Item)

#字典列表遍历
for i,j in Item:
     print(str(i)+":"+str(j))

#转换成列表
print(list(Item))

输出：
dict_items([('spanish', 'xibanya'), ('china', 'nb'), ('a', 1), ('usa', 'lj')])
spanish:xibanya
china:nb
a:1
usa:lj
[('spanish', 'xibanya'), ('china', 'nb'), ('a', 1), ('usa', 'lj')]

# pop()和popitem()

默认弹出最后一个位置的元素，pop返回末尾value，popitem返回一对key-value

#popitem()返回栈顶一对key，value可以
keyy,valuee=dic.popitem()
print(dic)
print(keyy,valuee)
print(dic.popitem())

结果
{'spanish': 'xibanya', 'china': 'zhongguo'}
a 1
('china', 'zhongguo') #返回元组

# setdefault

#setdefault提供了新增键值对，存在则不改变value，不存在就new
dic.setdefault('shit',12)
print(dic)
dic.setdefault('spanish',12)
print(dic)

输出：
{'spanish': 'xibanya', 'shit': 12}
{'spanish': 'xibanya', 'shit': 12}

'''
#encoding: utf-8
import progressbar
import requests
import sys
import time
from time import sleep
from tqdm import tqdm
# url='http://127.0.0.1/'

url = sys.argv[1]
list1=[]
length=0
d=[]

# 长度计算和列表导入

with open("php.txt", "r") as f1:
    for lines in f1:
        length = length + 1
        Lines=lines.strip()
        d.append(Lines)
# 进度条以长度作外循环，进行请求获取、筛选，并导入结果列表
for i in tqdm(range(1,length)):
            line=d[i]
            r=requests.get(url=url+line,headers={'Connection':'close'})
            if r.status_code == 200 or r.status_code == 403 :
                # print("url:"+r.url+" exist")
                    if r.url in list1:
                        pass
                    else:
                        list1.append(r.url)
            # print('finish')
            sleep(0.3)
# 结果输出
print("Finish and result is :\n")
for i in list1:
    print(i)
